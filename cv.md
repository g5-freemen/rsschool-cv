function duplicateCount(text){
  if (!text) return 0;
  let arr=text.toLowerCase().split('').sort(),
      doubles = [];
  for (let i=0; i<arr.length-1; i++) {
    if (arr[i]==arr[i+1]) doubles.push(arr[i]);
  }
  return new Set(doubles).size
}
